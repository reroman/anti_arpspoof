 47 /// Length in bytes of one Hardware Address.
 48 #define MAC_ADDR_LEN    6
 49 
 50 /// Length in bytes of one IP Address.

 54 #define IP_ONE      htonl( 1 )

 56 /// The maximum number of attempts to resolve a HW Address.
 57 #define MAX_TRIES_FOR_RESOLV    5

 70 /**
 71  * Stores a MAC Address
 72  */
 73 struct HWAddr{
 74     uint8_t hw[MAC_ADDR_LEN];
 75 
 76     /**
 77      * Creates an HWAddr object from an array of bytes with length size of 6.
 78      *
 79      * @param m The array with the 6 bytes of the HW Address.
 80      */
 81     HWAddr( const uint8_t m[] ){
 82         memcpy( hw, m, MAC_ADDR_LEN );
 83     }
 84 
 85     bool operator < ( const HWAddr &m ) const {
 86         return memcmp( hw, m.hw, MAC_ADDR_LEN ) < 0;
 87     }
 88 
 89     /**
 90      * Get a string representation of the HW Address in
 91      * the format xx:xx:xx:xx:xx:xx.
 92      *
 93      * @return The string representation of the HW Address.
 94      */
 95     string toString() const {
 96         ostringstream out;
 97 
 98         out << hex << setfill( '0' ) << setw( 2 );
 99         for( int i = 0 ; i < MAC_ADDR_LEN ; ){
100             out << static_cast<int>( hw[i] );
101             if( hw[i] == 0 )
102                 out << 0;
103             if( ++i != MAC_ADDR_LEN )
104                 out << ':';
105         }
106         return out.str();
107     }
108 
109 };

111 /**
112  * The representation of a ARP Frame including the 
113  * ethernet header.
114  */
115 struct ARPFrame{
116     uint8_t     eth_dst[MAC_ADDR_LEN];  ///< Ethernet destination address.
117     uint8_t     eth_src[MAC_ADDR_LEN];  ///< Ethernet source address.
118     uint16_t    eth_ethertype;          ///< Eheternet ethertype.
119     uint16_t    hw_type;                ///< ARP hardware type (Ethernet 0x0001).
120     uint16_t    protocol;               ///< ARP protocol (IP 0x0800).
121     uint8_t     hw_len;                 ///< ARP lenght in bytes of hardware address.
122     uint8_t     proto_len;              ///< ARP lenght in byte of protocol address.
123     uint16_t    opcode;                 ///< ARP Operation code (0x0001 for request, 0x0002 for reply).
124     uint8_t     hw_src[MAC_ADDR_LEN];   ///< ARP source HW address.
125     uint32_t    ip_src;                 ///< ARP source Protocol address.
126     uint8_t     hw_dst[MAC_ADDR_LEN];   ///< ARP target HW address.
127     uint32_t    ip_dst;                 ///< ARP target Protocol address.
128 } __attribute__((__packed__));
129 
130 /** Represents a key-value table.*/
131 typedef map< HWAddr, struct in_addr> ARPTable;

133 /** Stores some info about the netdevice */
134 struct LocalData{
135     int ifindex;                    ///< Index of the network interface.
136     uint32_t ipAddr;                ///< IP Address of the network interface.
137     uint32_t firstHost;             ///< IP Address of the first host in the network.
138     uint32_t lastHost;              ///< IP Address of the last host in the network.
139     uint8_t hwAddr[MAC_ADDR_LEN];   ///< Hawrdware Address of the network interface.
140 };

147 /**
148  * Get the local data of the netdevice given
149  * the interface name.
150  *
151  * @param ifname The name of the network interface
152  * @return A LocalData object with the local information.
153  *
154  * @throw runtime_error If some field couldn't be acquired.
155  */
156 LocalData loadLocalData( const char *ifname ) throw( runtime_error )
157 {
158     struct ifreq nic;
159     int sock = socket( AF_INET, SOCK_STREAM, 0 );
160     LocalData data;
161 
162     strncpy( nic.ifr_name, ifname, IFNAMSIZ-1 );
163     nic.ifr_name[IFNAMSIZ-1] = '\0';
164 
165     // Index
166     if( ioctl( sock, SIOCGIFINDEX, &nic ) < 0 ){
167         close( sock );
168         throw runtime_error( string(ifname) + ": " + string(strerror(errno)) );
169     }
170     data.ifindex = nic.ifr_ifindex;
171 
172     // IP Address
173     if( ioctl( sock, SIOCGIFADDR, &nic ) < 0 ){
174         close( sock );
175         throw runtime_error( "Getting IP Address: " + string(strerror(errno)) );
176     }
177     memcpy( &data.ipAddr, nic.ifr_addr.sa_data + 2, IP_ADDR_LEN );
178 
179     // Hw Address
180     if( ioctl( sock, SIOCGIFHWADDR, &nic ) < 0 ){
181         close( sock );
182         throw runtime_error( "Getting HW Address: " + string(strerror(errno)) );
183     }
184     memcpy( data.hwAddr, nic.ifr_netmask.sa_data, MAC_ADDR_LEN );
185 
186     // Broadcast for last host
187     if( ioctl( sock, SIOCGIFBRDADDR, &nic ) < 0 ){
188         close( sock );
189         throw runtime_error( "Getting Broadcast: " + string(strerror(errno)) );
190     }
191     memcpy( &data.lastHost, nic.ifr_broadaddr.sa_data + 2, IP_ADDR_LEN );
192 
193     // First host from the network IP
194     struct in_addr aux = { data.ipAddr };
195     data.firstHost = inet_netof( aux );
196     data.firstHost = (htonl( data.firstHost ) >> 8) + IP_ONE;
197 
198     return data;
199 }

201 /**
202  * Creates a socket for ARP frames.
203  *
204  * @param ifindex The network interface index to bind the socket.
205  * @return The socket descriptor.
206  *
207  * @throw runtime_error If the socket couldn't be opened (open raw sockets requires
208  * root provileges).
209  * @throw runtime_error  The maximum time to wait for a response couldn't be configured.
210  * @throw runtime_error socket could't bind to the interface.
211  */
212 int initSocket( int ifindex ) throw( runtime_error )
213 {
214     int sockfd;
215     struct sockaddr_ll sll;
216     struct timeval timer;
217 
218     if( (sockfd = socket( AF_PACKET, SOCK_RAW, htons(ETH_P_ARP) )) < 0 )
219         throw runtime_error( "socket: " + string(strerror(errno)) );
220 
221     timer.tv_sec = 0;
222     timer.tv_usec = 100000;
223     if( setsockopt( sockfd, SOL_SOCKET, SO_RCVTIMEO, &timer, sizeof(timer) ) < 0 )
224         throw runtime_error( strerror(errno) );
225 
226     memset( &sll, 0, sizeof(sll) );
227     sll.sll_family = AF_PACKET;
228     sll.sll_ifindex = ifindex;
229     sll.sll_protocol = htons( ETH_P_ARP );
230 
231     if( bind( sockfd, (struct sockaddr*) &sll, sizeof(sll) ) < 0 )
232         throw runtime_error( strerror(errno) );
233 
234     return sockfd;
235 }

237 /**
238  * Adds a permanent entry to the ARP cache of the system.
239  *
240  * @param ifname Name of the network interface.
241  * @param ip IP Address v4 of the new arp entry.
242  * @param hw An HWAddr object that contains the info about the hardware address.
243  *
244  * @throw runtime_error If the new entry couldn't be added.
245  * */
246 void addARPEntry(const char *ifname, struct in_addr ip, const HWAddr &hw)
247     throw( runtime_error )
248 {
249     struct arpreq arp;
250     int sock = socket( AF_INET, SOCK_DGRAM, 0 );
251 
252     arp.arp_pa.sa_family = AF_INET;
253     memcpy( arp.arp_pa.sa_data + 2, &ip.s_addr, IP_ADDR_LEN );
254     arp.arp_ha.sa_family = ARPHRD_ETHER;
255     memcpy( arp.arp_ha.sa_data, hw.hw, MAC_ADDR_LEN );
256     strncpy( arp.arp_dev, ifname, IFNAMSIZ - 1 );
257     arp.arp_dev[IFNAMSIZ - 1] = '\0';
258     arp.arp_flags = ATF_COM | ATF_PERM;
259 
260     if( ioctl( sock, SIOCSARP, &arp ) == -1 ){
261         close( sock );
262         throw runtime_error( "Add ARP entry: " + string(strerror(errno)) );
263     }
264     close( sock );
265 }

267 /**
268  * Makes a scan for ARP entries.
269  *
270  * @param sfd The ARP socket to send/receive ARP frames.
271  * @param ld An LocalData object that contains the local info about the
272  * network interface.
273  *
274  * @return ARPTable that contains the ARP entries in the network.
275  *
276  * @note The last host is not included in the scan.
277  * @see initSocket()
278  */
279 ARPTable scan( int sfd, const LocalData &ld )
280 {
281     struct in_addr host;
282     ARPTable table;
283     ARPFrame request, reply;
284     int attempts;
285 
286     // Setting constant values of the request
287     memset( request.eth_dst, 0xff, MAC_ADDR_LEN );
288     memcpy( request.eth_src, ld.hwAddr, MAC_ADDR_LEN );
289     request.eth_ethertype = htons( ETH_P_ARP );
290     request.hw_type = htons( ARPHRD_ETHER );
291     request.protocol = htons( ETH_P_IP );
292     request.hw_len = MAC_ADDR_LEN;
293     request.proto_len = IP_ADDR_LEN;
294     request.opcode = htons(ARPOP_REQUEST);
295     memcpy( request.hw_src, ld.hwAddr, MAC_ADDR_LEN );
296     request.ip_src = ld.ipAddr;
297     memset( request.hw_dst, 0, MAC_ADDR_LEN );
298 
299     // Bucle for hosts
300     for( host.s_addr = ld.firstHost ; host.s_addr != ld.lastHost ; host.s_addr += IP_ONE ){
301         request.ip_dst = host.s_addr;
302         attempts = MAX_TRIES_FOR_RESOLV;
303         cout << "Resolving " << inet_ntoa( host ) << '\r';
304         cout.flush();
305         write( sfd, &request, sizeof(request) ); // Send the request.
306         do{
307             if( read( sfd, &reply, sizeof(reply) ) > 0 ){ // Receive the reply
308                 // Verify the reply and sender.
309                 if( ntohs(reply.opcode) == ARPOP_REPLY && reply.ip_src == host.s_addr ){
310                     HWAddr hw( reply.hw_src );
311                     struct in_addr aux = { reply.ip_src };
312                     table[hw] = aux;
313                     attempts = 0;
314                 }
315                 else // Not the answer what we want.
316                     attempts--;
317             }
318             else // Some error or no response.
319                 attempts = 0;
320         }while( attempts );
321     }
322     cout << endl;
323     return table;
324 }

326 /**
327  * An infinite bucle that analyzes new ARP replies.
328  * The bucle stops setting ::active to false.
329  *
330  * @param sfd The ARP socket for receive ARP replies.
331  * @param ifname The name of the interface network.
332  * @param table The ARPTable that contains the ARP entries. 
333  *
334  * @throw runtime_error If a request of add ARP entry failed.
335  */
336 void guard( int sfd, const char *ifname, const ARPTable &table )
337 {
338     ARPFrame reply;
339     set<uint32_t> ignored;
340     string option;
341     bool find;
342 
343     while( active ){
344         // Receive the data
345         if( read( sfd, &reply, sizeof(reply) ) > 0 ){
346             // Verify the reply
347             if( ntohs(reply.opcode) == ARPOP_REPLY ){
348                 HWAddr hw( reply.hw_src );
349                 struct in_addr ip = { reply.ip_src };
350 
351                 try{
352                     struct in_addr reg = table.at(hw); // Check our ARP Table for the sender.
353 
354                     if( reg.s_addr != ip.s_addr &&  // If the MAC doesn't match with the IP
355                             ignored.find(ip.s_addr) == ignored.end() ){ // ... And it's not ignored
356                             find = true;
357 
358                             // Notice to the user
359                             cout << hw.toString() << " is poisoning " << inet_ntoa(ip) <<
360                                 ". Would you like to add a permanent entry to avoid the faking? (Y/N) ";
361                             getline( cin, option );
362 
363                             if( option != "N" && option != "n" ){
364                                 find = false;
365                                 for( auto &i : table ){ // Look for the IP Address, if it is.
366                                     if( i.second.s_addr == ip.s_addr ){
367                                         try{
368                                             addARPEntry( ifname, ip, i.first );
369                                             cout << "Entry added" << endl;
370                                             find = true;
371                                         }
372                                         catch( runtime_error &e ){
373                                             cerr << e.what() << endl;
374                                         }
375                                         break;
376                                     }
377                                 }
378                             }
379                             if( !find ) // The IP spoofed is not in out ARP Table
380                                 cout << "There's a missing entry. Please run the tool again for a new scan." << endl;
381                             ignored.insert( ip.s_addr );
382                         } // End if for ignoring
383                 }
384                 // The HW Address of the sender is not in our ARP Table
385                 catch( out_of_range ){
386                     cout << "There's a new device. You should try with a new scan." << endl;
387                 } // The received entry is not in our ARP Table
388             } // End if for replies ARP
389         } // End if for reading
390     } // End while
391 }

393 /**
394  * Kill signal handler. Change the value of ::active to stop
395  * the execution of guard().
396  */
397 void sigKill(int){
398     active = false;
399 }

401 /**
402  * Main function of the program.
403  *
404  * @param interface_name The name of the network interface to use.
405  */
406 int main( int argc, char **argv )
407 {
408     if( argc != 2 ){
409         cerr << "Uso:\n\t" << *argv << " interface_name" << endl;
410         return 1;
411     }
412 
413     int sockfd;
414     LocalData data;
415     ARPTable arpTable;
416 
417     try{
418         data = loadLocalData( argv[1] );
419         sockfd = initSocket( data.ifindex );
420     }
421     catch( runtime_error &e ){
422         cerr << e.what() << endl;
423         return 1;
424     }
425 
426 
427     arpTable = scan( sockfd, data );
428 
429     // Output the ARP table.
430     cout << arpTable.size() << " entries found. "
431         "If you think there's missing devices, please run the tool again.\n\n"
432         "\tHW Address\t\t\tIP Address\n";
433     for( auto &i : arpTable )
434         cout << '\t' << i.first.toString() << "\t\t" << inet_ntoa(i.second) << endl;
435 
436     cout << "\nAnalyzing ARP replies. Press CTRL-C to exit\n\n";
437     signal( SIGINT, sigKill );
438     guard( sockfd, argv[1], arpTable );
439 
440     cout << "\rClosing socket..." << endl;
441     close( sockfd );
442     return 0;
443 }
